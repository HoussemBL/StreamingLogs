

***********Kafka topic
docker-compose exec kafka  \
kafka-topics --create --topic logs1 --partitions 1 --replication-factor 1 --if-not-exists --zookeeper zookeeper:2181


***Hear from a topic
docker-compose exec kafka    kafka-console-consumer --bootstrap-server localhost:29092 --topic logs1 --from-beginning --max-messages 100


******** delete topic
docker-compose exec kafka  \
kafka-topics --zookeeper zookeeper:2181 --delete --topic apachelogs
